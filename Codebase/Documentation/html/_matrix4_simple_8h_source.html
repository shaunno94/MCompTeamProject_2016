81  //Multiplies 'this' matrix by matrix 'a'. Performs the multiplication in 'OpenGL' order (ie, backwards)
82  inline Matrix4Simple operator*(const Matrix4Simple& a) const {
83  Matrix4Simple out;
84  out.ToZero();
85  //Students! You should be able to think up a really easy way of speeding this up...
86  for(unsigned int r = 0; r < 4; ++r) {
87  for(unsigned int c = 0; c < 4; ++c) {
88  for(unsigned int i = 0; i < 4; ++i)
89  out.values[c + (r * 4)] += this->values[c + (i * 4)] * a.values[(r * 4) + i];
90  }
91  }
92  return out;
93  }
